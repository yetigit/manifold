29// Vector implementation optimized for managed memory, will perform memory
30// prefetching to minimize page faults and use parallel/GPU copy/fill depending
31// on data size. This will also handle builds without CUDA or builds with CUDA
32// but run without CUDA GPU properly.
33//
34// Note that the constructor and resize function will not perform initialization
35// if the parameter val is not set. Also, this implementation is a toy
36// implementation that did not consider things like non-trivial
37// constructor/destructor, please keep T trivial.
38template <typename T>
39class ManagedVec {
40 public:
41 typedef T *Iter;
42 typedef const T *IterC;

44 ManagedVec() {
45 size_ = 0;
46 capacity_ = 0;
47 onHost = true;
48 }

50 // note that we leave the memory uninitialized
51 ManagedVec(size_t n) {
52 size_ = n;
53 capacity_ = n;
54 onHost = autoPolicy(n) != ExecutionPolicy::ParUnseq;
55 if (n == 0) return;
56 mallocManaged(&ptr_, size_ * sizeof(T));
57 }
58
59 ManagedVec(size_t n, const T &val) {
60 size_ = n;
61 capacity_ = n;
62 if (n == 0) return;
63 auto policy = autoPolicy(n);
64 onHost = policy != ExecutionPolicy::ParUnseq;
65 mallocManaged(&ptr_, size_ * sizeof(T));
66 prefetch(ptr_, size_ * sizeof(T), onHost);
67 uninitialized_fill_n(policy, ptr_, n, val);
68 }

70 ~ManagedVec() {
71 if (ptr_ != nullptr) freeManaged(ptr_);
72 ptr_ = nullptr;
73 size_ = 0;
74 capacity_ = 0;
75 }

77 ManagedVec(const std::vector<T> &vec) {
78 size_ = vec.size();
79 capacity_ = size_;
80 auto policy = autoPolicy(size_);
81 onHost = policy != ExecutionPolicy::ParUnseq;
82 if (size_ != 0) {
83 mallocManaged(&ptr_, size_ * sizeof(T));
84 fastUninitializedCopy(ptr_, vec.data(), size_, policy);
85 }
86 }

88 ManagedVec(const ManagedVec<T> &vec) {
89 size_ = vec.size_;
90 capacity_ = size_;
91 auto policy = autoPolicy(size_);
92 onHost = policy != ExecutionPolicy::ParUnseq;
93 if (size_ != 0) {
94 mallocManaged(&ptr_, size_ * sizeof(T));
95 prefetch(ptr_, size_ * sizeof(T), onHost);
96 uninitialized_copy(policy, vec.begin(), vec.end(), ptr_);
97 }
98 }

100 ManagedVec(ManagedVec<T> &&vec) {

103 capacity_ = vec.capacity_;
104 onHost = vec.onHost;

106 vec.size_ = 0;
107 vec.capacity_ = 0;

111 if (&vec == this) return *this;
112 if (ptr_ != nullptr) freeManaged(ptr_);
113 size_ = vec.size_;
114 capacity_ = vec.size_;

116 onHost = policy != ExecutionPolicy::ParUnseq;
117 if (size_ != 0) {

119 prefetch(ptr_, size_ * sizeof(T), onHost);
120 uninitialized_copy(policy, vec.begin(), vec.end(), ptr_);

122 return *this;
123 }

143 void resize(size_t n, const T &val) {
144 reserve(n);
145 if (size_ < n) {
146 uninitialized_fill(autoPolicy(n - size_), ptr_ + size_, ptr_ + n, val);
147 }
148 size_ = n;
149 }

151 void reserve(size_t n) {
152 if (n > capacity_) {
153 T *newBuffer;
154 mallocManaged(&newBuffer, n * sizeof(T));
155 prefetch(newBuffer, size_ * sizeof(T), onHost);
156 if (size_ > 0) {
157 uninitialized_copy(autoPolicy(size_), ptr_, ptr_ + size_, newBuffer);
158 }
159 if (ptr_ != nullptr) freeManaged(ptr_);
160 ptr_ = newBuffer;
161 capacity_ = n;
162 }
163 }
164
165 void shrink_to_fit() {
166 T *newBuffer = nullptr;
167 if (size_ > 0) {
168 mallocManaged(&newBuffer, size_ * sizeof(T));
169 prefetch(newBuffer, size_ * sizeof(T), onHost);
170 uninitialized_copy(autoPolicy(size_), ptr_, ptr_ + size_, newBuffer);
171 }
172 freeManaged(ptr_);
173 ptr_ = newBuffer;
174 capacity_ = size_;
175 }
176
177 void push_back(const T &val) {
178 if (size_ >= capacity_) {
179 // avoid dangling pointer in case val is a reference of our array
180 T val_copy = val;
181 reserve(capacity_ == 0 ? 128 : capacity_ * 2);
182 onHost = true;
183 ptr_[size_++] = val_copy;
184 return;
185 }
186 ptr_[size_++] = val;
187 }

189 void swap(ManagedVec<T> &other) {
190 std::swap(ptr_, other.ptr_);
191 std::swap(size_, other.size_);
192 std::swap(capacity_, other.capacity_);
193 std::swap(onHost, other.onHost);
194 }

197 if (toHost != onHost) prefetch(ptr_, size_ * sizeof(T), toHost);

199 }

241 static constexpr int DEVICE_MAX_BYTES = 1 << 16;
242
243 static void mallocManaged(T **ptr, size_t bytes) {
244#ifdef MANIFOLD_USE_CUDA
245 if (CudaEnabled())
246 cudaMallocManaged(reinterpret_cast<void **>(ptr), bytes);
247 else
248#endif
249 *ptr = reinterpret_cast<T *>(malloc(bytes));
250 }
251
252 static void freeManaged(T *ptr) {
253#ifdef MANIFOLD_USE_CUDA
254 if (CudaEnabled())
255 cudaFree(ptr);
256 else
257#endif
258 free(ptr);
259 }
260
261 static void prefetch(T *ptr, int bytes, bool onHost) {
262#ifdef MANIFOLD_USE_CUDA
263 if (bytes > 0 && CudaEnabled())
264 cudaMemPrefetchAsync(ptr, std::min(bytes, DEVICE_MAX_BYTES),
265 onHost ? cudaCpuDeviceId : 0);
266#endif
267 }
268
269 // fast routine for memcpy from std::vector to ManagedVec
270 static void fastUninitializedCopy(T *dst, const T *src, int n,
271 ExecutionPolicy policy) {
272 prefetch(dst, n * sizeof(T), policy != ExecutionPolicy::ParUnseq);
273 switch (policy) {
274 case ExecutionPolicy::ParUnseq:
275#ifdef MANIFOLD_USE_CUDA
276 cudaMemcpy(dst, src, n * sizeof(T), cudaMemcpyHostToDevice);
277#endif
278 case ExecutionPolicy::Par:
279 thrust::uninitialized_copy_n(thrust::MANIFOLD_PAR_NS::par, src, n, dst);
280 break;
281 case ExecutionPolicy::Seq:
282 thrust::uninitialized_copy_n(thrust::cpp::par, src, n, dst);
283 break;
284 }
285 }

288/*
289 * Host and device vector implementation. This uses `thrust::universal_vector`
290 * for storage, so data can be moved by the hardware on demand, allows using
291 * more memory than the available GPU memory, reduce memory overhead and provide
292 * speedup due to less synchronization.
293 *
294 * Due to https://github.com/NVIDIA/thrust/issues/1690 , `push_back` operations
295 * on universal vectors are *VERY* slow, so a `std::vector` is used as a cache.
296 * The cache will be created when we perform `push_back` or `reserve` operations
297 * on the `VecDH`, and destroyed when we try to access device iterator/pointer.
298 * For better performance, please avoid interspersing `push_back` between device
299 * memory accesses, as that will cause repeated synchronization and hurts
300 * performance.
301 * Note that it is *NOT SAFE* to first obtain a host(device) pointer, perform
302 * some device(host) modification, and then read the host(device) pointer again
303 * (on the same vector). The memory will be inconsistent in that case.
304 */
305template <typename T>
306class VecDH {
307 public:
308 VecDH() {}

312 // the data is initialized.
