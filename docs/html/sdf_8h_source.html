79__host__ __device__ Uint64 SpreadBits3(Uint64 v) {
80 v = v & 0x1fffff;
81 v = (v | v << 32) & 0x1f00000000ffff;
82 v = (v | v << 16) & 0x1f0000ff0000ff;
83 v = (v | v << 8) & 0x100f00f00f00f00f;
84 v = (v | v << 4) & 0x10c30c30c30c30c3;
85 v = (v | v << 2) & 0x1249249249249249;
86 return v;
87}
88
89__host__ __device__ Uint64 SqueezeBits3(Uint64 v) {
90 v = v & 0x1249249249249249;
91 v = (v ^ v >> 2) & 0x10c30c30c30c30c3;
92 v = (v ^ v >> 4) & 0x100f00f00f00f00f;
93 v = (v ^ v >> 8) & 0x1f0000ff0000ff;
94 v = (v ^ v >> 16) & 0x1f00000000ffff;
95 v = (v ^ v >> 32) & 0x1fffff;
96 return v;
97}

99// This is a modified 3D MortonCode, where the xyz code is shifted by one bit
100// and the w bit is added as the least significant. This allows 21 bits per x,
101// y, and z channel and 1 for w, filling the 64 bit total.
102__host__ __device__ Uint64 MortonCode(const glm::ivec4& index) {
103 return static_cast<Uint64>(index.w) | (SpreadBits3(index.x) << 1) |
104 (SpreadBits3(index.y) << 2) | (SpreadBits3(index.z) << 3);
105}
106
107__host__ __device__ glm::ivec4 DecodeMorton(Uint64 code) {
108 glm::ivec4 index;
109 index.x = SqueezeBits3(code >> 1);
110 index.y = SqueezeBits3(code >> 2);
111 index.z = SqueezeBits3(code >> 3);
112 index.w = code & 0x1u;
113 return index;
114}

116struct GridVert {
117 float distance = NAN;
118 int edgeVerts[7] = {-1, -1, -1, -1, -1, -1, -1};
119
120 __host__ __device__ int Inside() const { return distance > 0 ? 1 : -1; }
121
122 __host__ __device__ int NeighborInside(int i) const {
123 return Inside() * (edgeVerts[i] < 0 ? 1 : -1);
124 }
125};

127template <typename Func>
128struct ComputeVerts {
129 glm::vec3* vertPos;
130 int* vertIndex;
131 HashTableD<GridVert, identity> gridVerts;
132 const Func sdf;
133 const glm::vec3 origin;
134 const glm::ivec3 gridSize;
135 const glm::vec3 spacing;
136 const float level;
137
138 inline __host__ __device__ glm::vec3 Position(glm::ivec4 gridIndex) const {
139 return origin +
140 spacing * (glm::vec3(gridIndex) + (gridIndex.w == 1 ? 0.0f : -0.5f));
141 }
142
143 inline __host__ __device__ float BoundedSDF(glm::ivec4 gridIndex) const {
144 const float d = sdf(Position(gridIndex)) - level;
145
146 const glm::ivec3 xyz(gridIndex);
147 const bool onLowerBound = glm::any(glm::lessThanEqual(xyz, glm::ivec3(0)));
148 const bool onUpperBound = glm::any(glm::greaterThanEqual(xyz, gridSize));
149 const bool onHalfBound =
150 gridIndex.w == 1 && glm::any(glm::greaterThanEqual(xyz, gridSize - 1));
151 if (onLowerBound || onUpperBound || onHalfBound) return glm::min(d, 0.0f);
152
153 return d;
154 }
155
156 inline __host__ __device__ void operator()(Uint64 mortonCode) {
157 if (gridVerts.Full()) return;
158
159 const glm::ivec4 gridIndex = DecodeMorton(mortonCode);
160
161 if (glm::any(glm::greaterThan(glm::ivec3(gridIndex), gridSize))) return;
162
163 const glm::vec3 position = Position(gridIndex);
164
165 GridVert gridVert;
166 gridVert.distance = BoundedSDF(gridIndex);
167
168 bool keep = false;
169 // These seven edges are uniquely owned by this gridVert; any of them
170 // which intersect the surface create a vert.
171 for (int i = 0; i < 7; ++i) {
172 glm::ivec4 neighborIndex = gridIndex + Neighbors(i);
173 if (neighborIndex.w == 2) {
174 neighborIndex += 1;
175 neighborIndex.w = 0;
176 }
177 const float val = BoundedSDF(neighborIndex);
178 if ((val > 0) == (gridVert.distance > 0)) continue;
179 keep = true;
180
181 const int idx = AtomicAdd(*vertIndex, 1);
182 vertPos[idx] =
183 (val * position - gridVert.distance * Position(neighborIndex)) /
184 (val - gridVert.distance);
185 gridVert.edgeVerts[i] = idx;
186 }
187
188 if (keep) gridVerts.Insert(mortonCode, gridVert);
189 }
190};

192struct BuildTris {
193 glm::ivec3* triVerts;
194 int* triIndex;
195 const HashTableD<GridVert, identity> gridVerts;
196
197 __host__ __device__ void CreateTri(const glm::ivec3& tri,
198 const int edges[6]) {
199 if (tri[0] < 0) return;
200 int idx = AtomicAdd(*triIndex, 1);
201 triVerts[idx] = {edges[tri[0]], edges[tri[1]], edges[tri[2]]};
202 }
203
204 __host__ __device__ void CreateTris(const glm::ivec4& tet,
205 const int edges[6]) {
206 const int i = (tet[0] > 0 ? 1 : 0) + (tet[1] > 0 ? 2 : 0) +
207 (tet[2] > 0 ? 4 : 0) + (tet[3] > 0 ? 8 : 0);
208 CreateTri(TetTri0(i), edges);
209 CreateTri(TetTri1(i), edges);
210 }
211
212 __host__ __device__ void operator()(int idx) {
213 Uint64 basekey = gridVerts.KeyAt(idx);
214 if (basekey == kOpen) return;
215
216 const GridVert& base = gridVerts.At(idx);
217 const glm::ivec4 baseIndex = DecodeMorton(basekey);
218
219 glm::ivec4 leadIndex = baseIndex;
220 if (leadIndex.w == 0)
221 leadIndex.w = 1;
222 else {
223 leadIndex += 1;
224 leadIndex.w = 0;
225 }
226
227 // This GridVert is in charge of the 6 tetrahedra surrounding its edge in
228 // the (1,1,1) direction (edge 0).
229 glm::ivec4 tet(base.NeighborInside(0), base.Inside(), -2, -2);
230 glm::ivec4 thisIndex = baseIndex;
231 thisIndex.x += 1;
232
233 GridVert thisVert = gridVerts[MortonCode(thisIndex)];
234
235 tet[2] = base.NeighborInside(1);
236 for (const int i : {0, 1, 2}) {
237 thisIndex = leadIndex;
238 --thisIndex[Prev3(i)];
239 // MortonCodes take unsigned input, so check for negatives, given the
240 // decrement.
241 GridVert nextVert = thisIndex[Prev3(i)] < 0
242 ? GridVert()
243 : gridVerts[MortonCode(thisIndex)];
244 tet[3] = base.NeighborInside(Prev3(i) + 4);
245
246 const int edges1[6] = {base.edgeVerts[0],
247 base.edgeVerts[i + 1],
248 nextVert.edgeVerts[Next3(i) + 4],
249 nextVert.edgeVerts[Prev3(i) + 1],
250 thisVert.edgeVerts[i + 4],
251 base.edgeVerts[Prev3(i) + 4]};
252 thisVert = nextVert;
253 CreateTris(tet, edges1);
254
255 thisIndex = baseIndex;
256 ++thisIndex[Next3(i)];
257 nextVert = gridVerts[MortonCode(thisIndex)];
258 tet[2] = tet[3];
259 tet[3] = base.NeighborInside(Next3(i) + 1);
260
261 const int edges2[6] = {base.edgeVerts[0],
262 edges1[5],
263 thisVert.edgeVerts[i + 4],
264 nextVert.edgeVerts[Next3(i) + 4],
265 edges1[3],
266 base.edgeVerts[Next3(i) + 1]};
267 thisVert = nextVert;
268 CreateTris(tet, edges2);
269
270 tet[2] = tet[3];
271 }
272 }
273};

321template <typename Func>
322inline Mesh LevelSet(Func sdf, Box bounds, float edgeLength, float level = 0,
323 std::optional<ExecutionPolicy> policy = std::nullopt) {
324 Mesh out;
325
326 const glm::vec3 dim = bounds.Size();
327 const float maxDim = std::max(dim[0], std::max(dim[1], dim[2]));
328 const glm::ivec3 gridSize(dim / edgeLength);
329 const glm::vec3 spacing = dim / (glm::vec3(gridSize));
330
331 const Uint64 maxMorton = MortonCode(glm::ivec4(gridSize + 1, 1));
332
333 // Parallel policies violate will crash language runtimes with runtime locks
334 // that expect to not be called back by unregistered threads. This allows
335 // bindings use LevelSet despite being compiled with MANIFOLD_PAR
336 // active (CUDA is already avoided when Func is a function ptr).
337 const auto pol =
338 (!policy.has_value() ||
339 (policy.value() == ExecutionPolicy::ParUnseq && !CudaEnabled()))
340 ? autoPolicy(maxMorton)
341 : policy.value();
342
343 int tableSize = glm::min(
344 2 * maxMorton, static_cast<Uint64>(10 * glm::pow(maxMorton, 0.667)));
345 HashTable<GridVert, identity> gridVerts(tableSize);
346 VecDH<glm::vec3> vertPos(gridVerts.Size() * 7);
347
348 while (1) {
349 VecDH<int> index(1, 0);
350 // avoid handing dynamic function pointers to CUDA
351 for_each_n_wrapper<Func>(
352 pol, maxMorton + 1,
353 ComputeVerts<Func>({vertPos.ptrD(), index.ptrD(), gridVerts.D(), sdf,
354 bounds.min, gridSize + 1, spacing, level}));
355
356 if (gridVerts.Full()) { // Resize HashTable
357 const glm::vec3 lastVert = vertPos[index[0] - 1];
358 const Uint64 lastMorton =
359 MortonCode(glm::ivec4((lastVert - bounds.min) / spacing, 1));
360 const float ratio = static_cast<float>(maxMorton) / lastMorton;
361 if (ratio > 1000) // do not trust the ratio if it is too large
362 tableSize *= 2;
363 else
364 tableSize *= ratio;
365 gridVerts = HashTable<GridVert, identity>(tableSize);
366 vertPos = VecDH<glm::vec3>(gridVerts.Size() * 7);
367 } else { // Success
368 vertPos.resize(index[0]);
369 break;
370 }
371 }
372
373 VecDH<glm::ivec3> triVerts(gridVerts.Entries() * 12); // worst case
374
375 VecDH<int> index(1, 0);
376 for_each_n(pol, countAt(0), gridVerts.Size(),
377 BuildTris({triVerts.ptrD(), index.ptrD(), gridVerts.D()}));
378 triVerts.resize(index[0]);
379
380 out.vertPos.insert(out.vertPos.end(), vertPos.begin(), vertPos.end());
381 out.triVerts.insert(out.triVerts.end(), triVerts.begin(), triVerts.end());
382 return out;
383}
