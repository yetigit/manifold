53inline HOST_DEVICE float sind(float x) {
54 if (!std::isfinite(x)) return sin(x);
55 if (x < 0.0f) return -sind(-x);
56 int quo;
57 x = remquo(fabs(x), 90.0f, &quo);
58 switch (quo % 4) {
59 case 0:
60 return sin(glm::radians(x));
61 case 1:
62 return cos(glm::radians(x));
63 case 2:
64 return -sin(glm::radians(x));
65 case 3:
66 return -cos(glm::radians(x));
67 }
68 return 0.0f;
69}

84inline HOST_DEVICE glm::mat4x3 RotateUp(glm::vec3 up) {
85 up = glm::normalize(up);
86 glm::vec3 axis = glm::cross(up, {0, 0, 1});
87 float angle = glm::asin(glm::length(axis));
88 if (glm::dot(up, {0, 0, 1}) < 0) angle = glm::pi<float>() - angle;
89 return glm::mat4x3(glm::rotate(glm::mat4(1), angle, axis));
90}

103inline HOST_DEVICE int CCW(glm::vec2 p0, glm::vec2 p1, glm::vec2 p2,
104 float tol) {
105 glm::vec2 v1 = p1 - p0;
106 glm::vec2 v2 = p2 - p0;
107 float area = v1.x * v2.y - v1.y * v2.x;
108 float base2 = glm::max(glm::dot(v1, v1), glm::dot(v2, v2));
109 if (area * area <= base2 * tol * tol)
110 return 0;
111 else
112 return area > 0 ? 1 : -1;
113}

208 MeshGL() = default;
209 MeshGL(const Mesh& mesh) {
210 numProp = 3;
211 vertProperties.resize(numProp * mesh.vertPos.size());
212 for (int i = 0; i < mesh.vertPos.size(); ++i) {
213 for (int j : {0, 1, 2}) vertProperties[3 * i + j] = mesh.vertPos[i][j];
214 }
215 triVerts.resize(3 * mesh.triVerts.size());
216 for (int i = 0; i < mesh.triVerts.size(); ++i) {
217 for (int j : {0, 1, 2}) triVerts[3 * i + j] = mesh.triVerts[i][j];
218 }
219 halfedgeTangent.resize(4 * mesh.halfedgeTangent.size());
220 for (int i = 0; i < mesh.halfedgeTangent.size(); ++i) {
221 for (int j : {0, 1, 2, 3})
222 halfedgeTangent[4 * i + j] = mesh.halfedgeTangent[i][j];

423class Quality {
424 private:
425 inline static int circularSegments_ = 0;
426 inline static float circularAngle_ = 10.0f;
427 inline static float circularEdgeLength_ = 1.0f;
428
429 public:
440 static void SetMinCircularAngle(float angle) {
441 if (angle <= 0) return;
442 circularAngle_ = angle;
443 }
444
454 static void SetMinCircularEdgeLength(float length) {
455 if (length <= 0) return;
456 circularEdgeLength_ = length;
457 }
458
468 static void SetCircularSegments(int number) {
469 if (number < 3 && number != 0) return;
470 circularSegments_ = number;
471 }
472
480 static int GetCircularSegments(float radius) {
481 if (circularSegments_ > 0) return circularSegments_;
482 int nSegA = 360.0f / circularAngle_;
483 int nSegL = 2.0f * radius * glm::pi<float>() / circularEdgeLength_;
484 int nSeg = fmin(nSegA, nSegL) + 3;
485 nSeg -= nSeg % 4;
486 return nSeg;
487 }
488};
